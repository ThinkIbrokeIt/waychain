Quick Start

package main

import (
  "github.com/waychain/waychain-go"
)

func main() {
  // Connect to WayChain node
  client := waychain.Connect("https://api.waychain.org")

  // Check Dox_Dev badge level for deployer
  level := client.Precompile("0x13").Call("checkLevel", deployerAddr)
  if level < 2 {
    return "Insufficient Dox_Dev level"
  }

  // Deploy contract (requires badge verification)
  tx := client.Deploy(contractCode, deployerAddr)
  fmt("Deployed: %s", tx.Hash())
}
